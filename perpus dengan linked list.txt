//fungsi insert buku di akhir
void insert_last_buku(list_buku & first, pointer pBaru)
{ pointer last;
last = first;

if (last == NULL)
last = pBaru;
else if (last->next == NULL)
last->next = pBaru;
else if (first->next != NULL){
do{
last = last->next;
}while (last->next != NULL);
last->next = pBaru;
}
}
